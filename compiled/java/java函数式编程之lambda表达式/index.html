func sort(nums []int, compare func (a, b int) int) {
    length := len(nums)
    for i := length - 1; i >= 0; i-- {
        for j := i; j + 1 < length; j++ {
            cur := nums[j]
            next := nums[j + 1]
            if compare(cur, next) > 0 {
                nums[j], nums[j + 1] = next, cur
            }
        }
    }
}
